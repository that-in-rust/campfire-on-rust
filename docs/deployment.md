## Local Development Setup

### Quick Start

```bash
# Clone repository
git clone https://github.com/that-in-rust/campfire-on-rust.git
cd campfire-on-rust

# Build and run
cargo build
cargo run

# Access application
open http://localhost:3000
```

### Docker Commands

```bash
# Build image
docker build -t campfire-rust .

# Run container
docker run -d \
  --name campfire \
  -p 3000:3000 \
  -v $(pwd)/data:/app/data \
  -e RUST_LOG=info \
  campfire-rust

# View logs
docker logs -f campfire

# Stop container
docker stop campfire
```

### Docker Compose Setup

```yaml
# docker-compose.yml
version: '3.8'

services:
  campfire:
    build: .
    ports:
      - "3000:3000"
    volumes:
      - ./data:/app/data
      - ./logs:/app/logs
    environment:
      - RUST_LOG=info
      - DATABASE_URL=sqlite:/app/data/campfire.db
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:3000/health"]
      interval: 30s
      timeout: 10s
      retries: 3

  # Optional: Reverse proxy
  nginx:
    image: nginx:alpine
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx.conf:/etc/nginx/nginx.conf
      - ./ssl:/etc/nginx/ssl
    depends_on:
      - campfire
    restart: unless-stopped
```

### Systemd Service (Bare Metal)

```ini
# /etc/systemd/system/campfire.service
[Unit]
Description=Campfire Rust Chat Application
After=network.target

[Service]
Type=simple
User=campfire
Group=campfire
WorkingDirectory=/opt/campfire
ExecStart=/opt/campfire/campfire-rust
Restart=always
RestartSec=5
Environment=RUST_LOG=info
Environment=DATABASE_URL=sqlite:/opt/campfire/data/campfire.db
Environment=BIND_ADDRESS=127.0.0.1:3000

# Security settings
NoNewPrivileges=true
PrivateTmp=true
ProtectSystem=strict
ProtectHome=true
ReadWritePaths=/opt/campfire/data /opt/campfire/logs

[Install]
WantedBy=multi-user.target
```
